### 2.1 Wave C：命名规则（最先落地，A/B 依赖）

```ts
/** 两位零填充：1 → "01" */
function pad2(n: number): string {
  return String(n).padStart(2, '0')
}

/**
 * v6.9：构建「序列 ID → 全局集号区间 [start, end]」映射。
 * 仅 one_to_many(短剧)/one_to_one(长剧) 产品有效；none(小说/剧本) 返回空 Map。
 *
 * 集数来源：
 *   - 短剧：sequences/<ID>.md 场景表数据行数（一集一场景）；解析失败回退 episodesPerSequence[0]
 *   - 长剧：固定 1（one_to_one）
 * 按 sequence_list 的 seqIds 顺序累计全局集号。
 */
private async buildEpisodeRangeMap(
  seqIds: string[],
): Promise<Map<string, [number, number]>> {
  const profile = this.profileLock
  const map = new Map<string, [number, number]>()
  if (!profile || profile.sequenceToEpisode === 'none') return map

  let cursor = 1 // 全局集号游标
  for (const seqId of seqIds) {
    let count = 1
    if (profile.sequenceToEpisode === 'one_to_many') {
      const beatsMd = await safeRead(this.fileManager, `sequences/${seqId}.md`)
      const rows = countTableDataRows(beatsMd) // 复用现有函数
      count = rows > 0 ? rows : (profile.episodesPerSequence?.[0] ?? 8)
    }
    map.set(seqId, [cursor, cursor + count - 1])
    cursor += count
  }
  return map
}

/**
 * v6.9：按产品解析序列正文落盘路径。
 *   短剧：chapters/E01-E12.md（全局集号区间）
 *   长剧：chapters/E05.md（单集）
 *   小说/剧本：chapters/<seqId>.md（不变）
 * 守 INV-3：用 profile.kind+sequenceToEpisode 分支，不加 ProductProfile 字段。
 */
private function resolveChapterPath(
  seqId: string,
  episodeRange: Map<string, [number, number]>,
): string {
  const profile = this.profileLock
  if (!profile || profile.sequenceToEpisode === 'none') {
    return `chapters/${seqId}.md`
  }
  const range = episodeRange.get(seqId)
  if (!range) return `chapters/${seqId}.md` // 兜底
  const [start, end] = range
  if (profile.sequenceToEpisode === 'one_to_many') {
    return `chapters/E${pad2(start)}-E${pad2(end)}.md`
  }
  // one_to_one（长剧）
  return `chapters/E${pad2(start)}.md`
}
```

> 注：`resolveChapterPath` 作为类方法需改为接收 `episodeRange` 参数（或读取实例字段）。本骨架示意逻辑，实现时 `runWriterSequencePipeline`/`runWriterBatchPipeline` 共用一份 `episodeRangeMap`（批量入口算一次，单序列精修入口算一次）。

### 2.2 Wave A：序列级写作 pipeline

```ts
/**
 * v6.9 单序列写作 pipeline（精修入口 / 批量并发循环体）。
 *
 * ① 建档（0 LLM）：按 resolveChapterPath 落骨架占位，UI 立即出卡片
 * ② 分段产出：
 *    - 短剧(proseSplitUnit=episode)：逐集循环调 writer LLM（每集 1 次），
 *      <current_draft>=已产各集累积，集号 writer 内部 ## 第N集 自管理
 *    - 其余(none/scene)：单次调 writer LLM 产整序列（沿用现有直发逻辑）
 * ③ 收口：覆写落盘 + extractBehaviorTrack/extractForeshadowingState/updateBatchProgress
 *
 * 守 INV-5：自包含，只读 profileLock/fileManager，只写本序列 chapter 文件，
 *          不读不写他序列产物（并行硬前提）。
 */
private async runWriterSequencePipeline(
  subagent: SubagentSpec,
  seqId: string,
  instruction: string,
  history: ConversationTurn[] | undefined,
  episodeRange: Map<string, [number, number]>,
): Promise<ToolResult> {
  const skill = selectSkill(subagent.id, seqId) // 单 skill 直选
  const chapterPath = this.resolveChapterPath(seqId, episodeRange)
  const splitUnit = this.profileLock?.proseSplitUnit ?? 'none'
  const seqBeatsDoc = await safeRead(this.fileManager, `sequences/${seqId}.md`)

  // ① 建档骨架（0 LLM）
  await this.fileManager.writeFile(chapterPath, this.buildChapterSkeleton(seqId, chapterPath))
  this.emit('tool_complete', {
    toolId: subagent.id, toolName: subagent.name, skillId: skill.skillId, skillName: skill.name,
    writes: [chapterPath], success: true, output: '', message: `[${seqId}] 建档骨架已落盘`,
  })

  // ② 分段产出
  let finalDraft = ''
  if (splitUnit === 'episode') {
    // 短剧逐集循环：集数 = episodeRange 区间长度
    // v6.9 fix: 引擎侧累积——writer 只输出本集，引擎追加到 finalDraft，
    //           不依赖 writer 回填历史集（避免 LLM 偶发不回填致覆写丢集）
    const [start, end] = episodeRange.get(seqId) ?? [1, 8]
    const episodeCount = end - start + 1
    for (let i = 0; i < episodeCount; i++) {
      const epDraft = await this.runWriterStep(
        subagent, skill, seqId, chapterPath, seqBeatsDoc, finalDraft, instruction, history,
      )
      if (epDraft == null) {
        finalDraft += `\n\n<!-- 待补第${i + 1}集（生成失败） -->\n` // 降级不阻断
      } else {
        finalDraft = finalDraft ? `${finalDraft}\n\n${epDraft}` : epDraft // 引擎追加本集
      }
    }
  } else {
    // 长剧/小说/剧本：单次产整序列
    const draft = await this.runWriterStep(
      subagent, skill, seqId, chapterPath, seqBeatsDoc, '', instruction, history,
    )
    finalDraft = draft ?? `<!-- 待补正文 ${seqId}（生成失败） -->`
  }

  // ③ 收口落盘 + 运行时记忆
  await this.fileManager.writeFile(chapterPath, finalDraft)
  this.extractBehaviorTrack(seqId, finalDraft)
  this.extractForeshadowingState(seqId, finalDraft)
  this.updateBatchProgress(seqId, finalDraft)
  const softWarnings = this.runSoftValidation(finalDraft)

  return {
    success: true,
    writes: [chapterPath],
    output: '',
    skillId: skill.skillId,
    skillName: skill.name,
    warnings: softWarnings.length > 0 ? softWarnings : undefined,
  }
}

/**
 * v6.9 单步 writer LLM 调用（短剧逐集 / 单序列共用）。
 * 组装上下文（含 <current_draft>/<current_sequence_beats>/<product_profile>/...）
 * → sendMessage ×≤MAX_RETRIES → validateOutput → 返回 extracted[chapterPath]。
 * 复用现有 executeTool WRITER 分支的上下文组装逻辑，抽成可复用方法。
 */
private async runWriterStep(
  subagent: SubagentSpec, skill: SkillSpec, seqId: string, chapterPath: string,
  seqBeatsDoc: string, currentDraft: string,
  instruction: string, history: ConversationTurn[] | undefined,
): Promise<string | null> {
  // ... 复用现有 WRITER 分支的 same_act_sequences/prev_chapter/behaviorTrack/profile/foreshadow state 组装
  const context = appendExtraLabels(assembleContext(skill.reads, await this.readSkillFiles(skill)), [
    { label: 'current_draft', content: currentDraft },
    { label: 'current_target', content: currentDraft },
    { label: 'current_sequence_beats', content: seqBeatsDoc },
    // ... same_act_sequences / previous_chapter_draft / character_behavior_tracking
    { label: 'product_profile', content: this.profileLock ? renderProductProfileXml(this.profileLock) : '' },
    { label: 'shot_breakdown_spec', content: this.profileLock?.kind === 'short_drama' ? SHORT_DRAMA_SHOT_SPEC : '' },
    { label: 'foreshadowing_state', content: this.renderForeshadowingStateXml() },
    { label: 'batch_progress', content: this.renderBatchProgressXml() },
  ])
  const systemPrompt = skill.preamble ? `${skill.preamble}\n\n${skill.body}` : skill.body
  const specView: SkillSpec = { ...skill, writes: [chapterPath] } // effectiveWrites 替换 placeholder
  let userContent = buildAgentPrompt(context, instruction, history)

  for (let attempt = 0; attempt < MAX_RETRIES; attempt++) {
    try {
      const out = await this.sendWithRateLimit(systemPrompt, userContent) // Wave B 429 退避
      const vr = validateOutput(out, specView)
      if (vr.valid) return vr.extracted[chapterPath] ?? ''
      if (attempt < MAX_RETRIES - 1) {
        this.emit('tool_retry', { /* ... */ })
        const feedback = vr.structuralError ? `⚠️ 结构错误：${vr.structuralError}` : `⚠️ 格式错误...`
        userContent = `${userContent}\n\n---\n${feedback}`
      }
    } catch { if (attempt === MAX_RETRIES - 1) break }
  }
  return null
}
```

### 2.3 Wave B：多序列并发批量

```ts
/** 全序列写作并发池上限。默认 3（保守防 429）。 */
const WRITER_CONCURRENCY = 3

/**
 * v6.9 通用有界并发池：最多 limit 个 worker 同时跑，超出排队。
 * out[i] 与 items[i] 下标对齐（无论完成顺序）。
 * 供 writer 批量用；v6.7.5 scene_beats 并发可复用本方法（本专项不动其调用点）。
 */
private async runWithConcurrency<T, R>(
  items: T[], limit: number, worker: (item: T) => Promise<R>,
): Promise<R[]> {
  const out: R[] = new Array(items.length)
  let next = 0
  const runners = Array.from({ length: Math.min(limit, items.length) }, async () => {
    while (next < items.length) {
      const i = next++
      out[i] = await worker(items[i])
    }
  })
  await Promise.all(runners)
  return out
}

/**
 * v6.9 全序列并发批量写作：读 sequence_list 解析全部 seqIds + buildEpisodeRangeMap
 * → runWithConcurrency 并发跑 runWriterSequencePipeline → 汇总单 ToolResult。
 */
private async runWriterBatchPipeline(
  subagent: SubagentSpec,
  instruction: string,
  history?: ConversationTurn[],
): Promise<ToolResult> {
  const seqListMd = await safeRead(this.fileManager, 'sequence_list.md')
  const seqIds = this.parseSequenceIds(seqListMd)
  if (seqIds.length === 0) {
    return { success: false, error: '未能从 sequence_list.md 解析出任何序列 ID', skillName: subagent.name }
  }
  const episodeRange = await this.buildEpisodeRangeMap(seqIds)

  this.emit('tool_start', {
    toolId: subagent.id, toolName: subagent.name,
    message: `并发批量写作 ${seqIds.length} 个序列（并发 ${WRITER_CONCURRENCY}）`,
  })

  const results = await this.runWithConcurrency(seqIds, WRITER_CONCURRENCY,
    (seqId) => this.runWriterSequencePipeline(subagent, seqId, instruction, history, episodeRange))

  // 汇总（按下标对应，不依赖完成顺序）
  const writes: string[] = []
  const warnings: string[] = []
  let okCount = 0
  results.forEach((r, i) => {
    if (r.success) {
      okCount++
      if (r.writes) writes.push(...r.writes)
      if (r.warnings) warnings.push(...r.warnings)
    } else {
      warnings.push(`序列 ${seqIds[i]} 失败：${r.error}`)
    }
  })
  return {
    success: okCount > 0,
    writes,
    output: '',
    skillName: subagent.name,
    error: okCount === 0 ? '全部序列写作失败' : undefined,
    warnings: warnings.length > 0 ? warnings : undefined,
  }
}

/**
 * v6.9 429 指数退避重试（复用 classifyLLMError 识别 rate_limit）。
 * 非 429 直接抛交原流程；退避不消耗格式重试预算（MAX_RETRIES 专留格式错误）。
 */
private async sendWithRateLimit(systemPrompt: string, userContent: string): Promise<string> {
  const MAX_429_RETRIES = 3
  for (let i = 0; i < MAX_429_RETRIES; i++) {
    try {
      return await this.llm.sendMessage(systemPrompt, userContent)
    } catch (e) {
      if (classifyLLMError(e).type !== 'rate_limit' || i === MAX_429_RETRIES - 1) throw e
      const backoff = 1000 * Math.pow(2, i) // 1s → 2s → 4s
      this.emit('tool_retry', {
        toolId: '', toolName: '', skillId: '', skillName: '',
        attempt: i + 1, maxAttempts: MAX_429_RETRIES,
        message: `429 限流，${backoff}ms 后重试 (${i + 1}/${MAX_429_RETRIES})`,
      })
      await new Promise((r) => setTimeout(r, backoff))
    }
  }
  throw new Error('unreachable')
}
```

### 2.6 Wave F：assetStore.computeLabel 优化

```ts
function computeLabel(path: string): string {
  if (FILE_LABELS[path]) return FILE_LABELS[path]
  const seqMatch = path.match(/^sequences\/(.+)\.md$/)
  if (seqMatch) return `序列 ${seqMatch[1]}`
  const chMatch = path.match(/^chapters\/(.+)\.md$/)
  if (chMatch) {
    const name = chMatch[1]
    // v6.9：E01-E12 → "第1-12集"，E05 → "第5集"
    const range = name.match(/^E(\d+)-E(\d+)$/)
    if (range) return `第${Number(range[1])}-${Number(range[2])}集`
    const single = name.match(/^E(\d+)$/)
    if (single) return `第${Number(single[1])}集`
    return `章节 ${name}`
  }
  return path.replace(/\.md$/, '')
}
```

### 2.8 v6.9 bugfix：逐集覆写丢集（已落地 v6.7 现状）

**现象**：短剧逐集续写时，新集覆盖旧集，单个资产文件永远只剩 1 集。

**根因**：旧约定要求 writer 每次输出「`<current_draft>` 既有各集 + 本次新集」整篇，引擎整篇覆写。LLM 偶发只输出新集 → 历史集丢失。

**修复**（已落地 [orchestratorEngine.ts](../../src/orchestrator/orchestratorEngine.ts) + 短剧/长剧 writer SKILL）：
- writer SKILL 输出约定改为「只输出本单元」（短剧本集 / 长剧本场景段），不回填历史
- 引擎落盘改为追加模式：`proseSplitUnit != 'none'` 且既有草稿非空时，`writeFile(currentDraft + '\n\n' + 本单元)`
- `updateBatchProgress` 改基于累积后内容算 `lastUnit`（旧逻辑基于 writer 本次 output 会恒记 1）

> 此 fix 已在 v6.7 现状代码落地，v6.9 pipeline 沿用同一累积范式（2.2 逐集循环 `finalDraft` 引擎追加）。

## 4. 风险与缓解

| 风险 | 依据 | 缓解 |
|---|---|---|
| 429 限流 | 并发 3 × 多序列 × 短剧逐集 LLM 瞬时速率高 | `WRITER_CONCURRENCY=3` + `sendWithRateLimit` 指数退避 |
| 短剧逐集循环某集失败 | writer 单集 LLM 偶发格式错误 | 降级占位 `<!-- 待补第N集 -->` 不阻断；可单独精修 |
| 全局集号误算 | sequences 场记残缺致场景行数≠集数 | 解析失败回退 `episodesPerSequence[0]`；写作期场记已冻结达标 |
| 旧短剧 `chapters/S1-1.md` 残留 | v6.6/v6.7 旧文件与新 `E01-Exx.md` 共存 | 短剧首次 v6.9 写作时，`runWriterSequencePipeline` 建档前按 seqId 删除旧 `chapters/<seqId>.md`（若存在且与新路径不同） |
| 并发事件交错致 UI 错乱 | 多序列 tool_complete 交替到达 | emit 单回调同步；`chatStore` 按 `event.writes` 路径精准刷新 |
| 单序列失败拖垮整批 | 并发下某序列 abort | `runWriterSequencePipeline` 返回 ToolResult 不抛；失败进 `warnings` |
